\section*{dottimeline Quick Reference}

\begin{multicols}{2}

\subsection*{Basic Structure}
\begin{verbatim}
\begin{timeline}[options]
  \milestone[above]{date}{label}
  \milestone[below]{date}{label}
  \highlightmilestone{date}{label}
\end{timeline}
\end{verbatim}

\subsection*{Environment Options}
\begin{verbatim}
width=14           % Timeline width (cm)
heightabove=1.5    % Height above line (cm)
heightbelow=1.5    % Height below line (cm)
rotation=0         % Label angle (0,45,90)
labelwidth=3cm     % Label text width
fontsize=\small    % Label font size
startdate=YYYY-MM-DD
enddate=YYYY-MM-DD
color=timelineblue
highlightcolor=timelineorange
\end{verbatim}

\subsection*{Per-Milestone Options}
\begin{verbatim}
\milestone[above][heightabove=2.5]{date}{label}
\milestone[below][heightbelow=2.0]{date}{label}
\milestone[above][height=3.0]{date}{label}
\highlightmilestone[above][heightabove=2.5]{date}{label}
\end{verbatim}

\subsection*{Other Commands}
\begin{verbatim}
\timelinetitle{Title}
\datemarker{date}{label}
\milestoneat[above]{x}{date}{label}
\end{verbatim}

\subsection*{Examples}

\textbf{Standard:}
\begin{timeline}[width=8,heightabove=1,heightbelow=1]
  \milestone{2025-01-01}{Start}
  \milestone[below]{2025-06-01}{Mid}
  \milestone{2025-12-01}{End}
\end{timeline}
